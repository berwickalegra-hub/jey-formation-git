// frontend/src/lib/server/observability/request-context.ts
import 'server-only';
import { AsyncLocalStorage } from 'node:async_hooks';
import { randomUUID } from 'node:crypto';

export interface RequestContext {
  requestId: string;
  startedAt: number;
}

const als = new AsyncLocalStorage<RequestContext>();

export function makeRequestContext(headers: Headers): RequestContext {
  const inbound = headers.get('x-request-id');
  const requestId =
    inbound && /^[0-9a-f-]{8,64}$/i.test(inbound) ? inbound : randomUUID();
  return { requestId, startedAt: Date.now() };
}

export function withRequestContext<T>(
  ctx: RequestContext,
  fn: () => Promise<T> | T,
): Promise<T> | T {
  return als.run(ctx, fn);
}

export function getRequestId(): string | undefined {
  return als.getStore()?.requestId;
}

export function getRequestContext(): RequestContext | undefined {
  return als.getStore();
}
```

import 'server-only';
import { createLogger, type CreateLoggerOptions, type Logger } from '@/lib/server/logger';
import { getRequestId } from './request-context';

export function createRequestLogger(options: CreateLoggerOptions = {}): Logger {
  const base = createLogger(options);
  function decorate(ctx?: Record<string, unknown>): Record<string, unknown> {
    const requestId = getRequestId();
    if (!requestId) return ctx ?? {};
    return { ...(ctx ?? {}), requestId };
  }
  return {
    debug: (msg, ctx) => base.debug(msg, decorate(ctx)),
    info: (msg, ctx) => base.info(msg, decorate(ctx)),
    warn: (msg, ctx) => base.warn(msg, decorate(ctx)),
    error: (msg, ctx) => base.error(msg, decorate(ctx)),
  };
}

export const log: Logger = createRequestLogger();
```

    // Source: composed from D-12, D-14, and node:async_hooks documentation.
    // No Next.js types imported ã module is unit-testable without booting Next.
    import 'server-only';
    import { AsyncLocalStorage } from 'node:async_hooks';
    import { randomUUID } from 'node:crypto';

    export interface RequestContext {
      requestId: string;
      startedAt: number;
    }

    const als = new AsyncLocalStorage<RequestContext>();

    /**
     * Read X-Request-Id from inbound Headers, or mint a new one. Returns the
     * RequestContext but does NOT enter the ALS scope ã call `withRequestContext`
     * for that, or use `runWithRequestContext` from a route handler.
     */
    export function makeRequestContext(headers: Headers): RequestContext {
      const inbound = headers.get('x-request-id');
      // Accept inbound IDs only if they look UUID-shaped (defensive ã clients
      // shouldn't be able to inject log-spammable garbage).
      const requestId =
        inbound && /^[0-9a-f-]{8,64}$/i.test(inbound) ? inbound : randomUUID();
      return { requestId, startedAt: Date.now() };
    }

    /**
     * Run `fn` inside an ALS scope carrying the given context. Use this in route
     * handlers ã wrap the entire handler body so all downstream awaits see the
     * same context.
     */
    export function withRequestContext<T>(
      ctx: RequestContext,
      fn: () => Promise<T> | T,
    ): Promise<T> | T {
      return als.run(ctx, fn);
    }

    /** Fetch the current request ID, or undefined if outside any context. */
    export function getRequestId(): string | undefined {
      return als.getStore()?.requestId;
    }

    /** Fetch the entire current context, or undefined. */
    export function getRequestContext(): RequestContext | undefined {
      return als.getStore();
    }
    ```

        const ctx = makeRequestContext(new Headers());
        const after = Date.now();
        expect(ctx.startedAt).toBeGreaterThanOrEqual(before);
        expect(ctx.startedAt).toBeLessThanOrEqual(after);
      });

        expect(seen).toEqual(ctx);
      });

        expect(getRequestId()).toBeUndefined();

    import 'server-only';
    import { createLogger, type CreateLoggerOptions, type Logger } from '@/lib/server/logger';
    import { getRequestId } from './request-context';

    /**
     * Wraps the existing logger so every log line picks up the current request ID
     * from AsyncLocalStorage. The base logger is created lazily so callers can
     * still pass options if they want a non-default config.
     */
    export function createRequestLogger(options: CreateLoggerOptions = {}): Logger {
      const base = createLogger(options);
      function decorate(ctx?: Record<string, unknown>): Record<string, unknown> {
        const requestId = getRequestId();
        if (!requestId) return ctx ?? {};
        return { ...(ctx ?? {}), requestId };
      }
      return {
        debug: (msg, ctx) => base.debug(msg, decorate(ctx)),
        info: (msg, ctx) => base.info(msg, decorate(ctx)),
        warn: (msg, ctx) => base.warn(msg, decorate(ctx)),
        error: (msg, ctx) => base.error(msg, decorate(ctx)),
      };
    }

    /** Default singleton ã most call sites use this. */
    export const log: Logger = createRequestLogger();
    ```

    import { makeRequestContext, withRequestContext } from './request-context';

    describe('createRequestLogger (OBS-04)', () => {

      let errorSpy: ReturnType<typeof vi.spyOn>;
      const log = createRequestLogger({ env: 'development' });

        // Capture console output. The base logger writes to console.log for
        // info/debug/warn and console.error for error.

        errorSpy.mockRestore();

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Inbound HTTP request ã request-context module | Untrusted client controls the `X-Request-Id` header; module validates against strict regex before reuse. |
| ALS scope ã logger emit | Every log line crosses the ALS boundary at emit time; missing context degrades gracefully (no requestId in line). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-0-01 | Tampering / Repudiation | Inbound `X-Request-Id` header poisoning | mitigate | Validate against `/^[0-9a-f-]{8,64}$/i` before reuse; fall back to `randomUUID()` on mismatch (Pitfall 3 of RESEARCH.md). Rejected values do NOT reach the logger or response header. Severity: low (defense-in-depth; log-poisoning impact is limited to log readability, not auth/authz). |
| T-ctx-01 | Information Disclosure | requestId leaking PII via inbound header reuse | accept | The regex `^[0-9a-f-]{8,64}$/i` already constrains the character set to UUID-shape ã no PII (e.g. emails, phone numbers, names) can survive validation. Severity: very low. |
| T-ctx-02 | Denial of Service | ALS context retention causing memory growth | accept | `AsyncLocalStorage` retains its store only for the duration of `als.run(ctx, fn)`. Once the wrapped Promise resolves, the store is released. Severity: very low (Node primitive; no leak path). |
| T-log-01 | Information Disclosure | requestId field in production logs revealing access patterns | accept | requestId is opaque randomness (UUID v4) ã no metadata leaks. Severity: very low. |
| T-log-02 | Tampering | Wrapper bypasses redaction in `lib/server/logger.ts` | mitigate | Wrapper calls `base.{level}` (the existing logger), which performs redaction. The wrapper only ADDS `requestId` to the ctx object before delegating; it does not bypass the base logger's redaction step. Severity: low. |
